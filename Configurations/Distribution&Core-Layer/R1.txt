!Configuring OSPF + advertising default gateway to the L3 switches + changing the reference value to 10G + Setting the network type is P-t-P!

!Configuring PAT + making it use the Router's Interface IP!

exit

ip nat outside

int range g0/0/1 - 2

!Configure a standard ACL so only IT and Server-Room can SSH!
access-list 10 permit 192.168.40.0 0.0.0.255
ip access-list standard 10
20 permit 192.168.70.0 0.0.0.255
30 deny any
remark Only IT and Server's IPs can SSH
exit

line vty 0 15
ip access-class 10 in
exit


!Configuring Extended ACL for blocking traffic coming from OUTSIDE!
ip access-list extended block_outside
remark no traffic from the internet shall pass
10 deny ip 192.168.0.0 0.0.255.255 any
20 deny ip 172.16.0.0 0.15.255.255 any
30 deny ip 10.0.0.0 0.255.255.255 any
40 deny ip 127.0.0.0 0.255.255.255 any

remark allow only replies to connections from INSIDE
50 permit icmp any any echo-reply
60 permit icmp any any unreachable
70 permit icmp any any time-exceeded
80 permit tcp any any established

remark deny and log everything else trying to get in
90 deny ip any any log
exit

int g0/0/0
ip access-group block_outside in
exit
